CREATE OR REPLACE FUNCTION         FU_BMC_GETPROVFORN (
    P_DTREF                 DATE,
    P_CODFRUTA              INT,
    P_CODPARCFORNECEDOR     INT,
    P_CODGRUPOPRODVARIEDADE INT,
    P_CODPARCCLIENTE        INT,
    P_MERCADO              VARCHAR,
    P_CODPROJ               INT,
    P_VLRRECEITALIQ         NUMBER,
    P_VLRCUSTOCALCULADO     NUMBER,
    P_VLRCOMISSAOCOM        NUMBER,
    P_CODPAIS               INT,
    P_NRROMANEIO            INT
) RETURN FLOAT DETERMINISTIC RESULT_CACHE RELIES_ON (AD_CONFPROVFORN, AD_CONFPROVFORNREG) IS
    V_VLRPROVFORN FLOAT;
BEGIN
    SELECT CASE WHEN r.PROVISIONAR ='N' THEN 0 ELSE ROUND(
               COALESCE(
                   (P_VLRRECEITALIQ - P_VLRCUSTOCALCULADO - P_VLRCOMISSAOCOM) +
                   NVL(r.VALORAJUSTEADICIONAL, 0),
                   0
               ),
               2) END
          
    
    INTO V_VLRPROVFORN
    FROM AD_CONFPROVFORN c
    LEFT JOIN AD_CONFPROVFORNREG r 
        ON c.CODCONFPROVFORN = r.CODCONFPROVFORN
        AND TRUNC(P_DTREF) BETWEEN TRUNC(c.DTVIGENCIAINICIO) AND TRUNC(c.DTVIGENCIAFINAL)
        AND (r.codproj = P_CODPROJ OR r.codproj IS NULL)
        AND (r.codfruta = P_CODFRUTA OR r.codfruta IS NULL)
        AND (r.CODGRUPOPROD = P_CODGRUPOPRODVARIEDADE OR r.CODGRUPOPROD IS NULL)
        AND (r.MERCADO = P_MERCADO OR r.MERCADO IS NULL)
        AND (r.CODPARCFORNECEDOR = P_CODPARCFORNECEDOR OR r.CODPARCFORNECEDOR IS NULL)
        AND (r.CODPARCCLIENTE = P_CODPARCCLIENTE OR r.CODPARCCLIENTE IS NULL)
        AND (r.CODPAIS = P_CODPAIS OR r.CODPAIS IS NULL)
        AND (r.NRROMANEIO = P_NRROMANEIO OR r.NRROMANEIO IS NULL)
       --  AND NVL(r.PROVISIONAR, 'S') = 'S' -- Prioriza 'S' em caso de nulo
    ORDER BY 
        -- Priorização das regras de negócio, da mais específica para a mais geral
        CASE 
            WHEN r.codproj IS NOT NULL AND r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL AND r.CODPARCFORNECEDOR IS NOT NULL AND r.CODPARCCLIENTE IS NOT NULL AND r.PROVISIONAR = 'N' THEN 1 
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL AND r.CODPARCCLIENTE IS NOT NULL AND r.PROVISIONAR = 'N' THEN 2
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL AND r.CODPARCFORNECEDOR IS NOT NULL  AND r.PROVISIONAR = 'N' THEN 3
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL AND r.NRROMANEIO IS NOT NULL  AND r.PROVISIONAR = 'N' THEN 4
            WHEN r.codproj IS NOT NULL AND r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL AND r.CODPARCFORNECEDOR IS NOT NULL  AND r.PROVISIONAR = 'N' THEN 5
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL  AND r.PROVISIONAR = 'N' THEN 6
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.PROVISIONAR = 'N'  THEN 7
            WHEN r.codfruta IS NOT NULL AND r.CODPARCFORNECEDOR IS NOT NULL AND r.PROVISIONAR = 'N' THEN 8
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.CODPAIS IS NOT NULL AND r.PROVISIONAR = 'N' THEN 9
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.CODPROJ IS NOT NULL AND R.CODPARCFORNECEDOR IS NOT NULL AND r.PROVISIONAR = 'N' THEN 10
            WHEN r.codfruta IS NOT NULL AND r.PROVISIONAR = 'N'  THEN 11
            WHEN r.codfruta IS NOT NULL AND r.MERCADO IS NOT NULL  AND r.PROVISIONAR = 'N' THEN 12
            WHEN r.codproj IS NOT NULL AND r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL AND r.CODPARCFORNECEDOR IS NOT NULL AND r.CODPARCCLIENTE IS NOT NULL AND NVL(r.PROVISIONAR, 'N') = 'S' THEN 13
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL AND r.CODPARCCLIENTE IS NOT NULL AND NVL(r.PROVISIONAR, 'N') = 'S' THEN 14
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL AND r.CODPARCFORNECEDOR IS NOT NULL  AND NVL(r.PROVISIONAR, 'N') = 'S' THEN 15
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL AND r.NRROMANEIO IS NOT NULL  AND NVL(r.PROVISIONAR, 'N') = 'S' THEN 16
            WHEN r.codproj IS NOT NULL AND r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL AND r.CODPARCFORNECEDOR IS NOT NULL  AND NVL(r.PROVISIONAR, 'N') = 'S' THEN 17
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.MERCADO IS NOT NULL  AND NVL(r.PROVISIONAR, 'N') = 'S' THEN 18
            WHEN r.codfruta IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND NVL(r.PROVISIONAR, 'N') = 'S'  THEN 19
            WHEN r.codfruta IS NOT NULL AND r.CODPARCFORNECEDOR IS NOT NULL  AND NVL(r.PROVISIONAR, 'N') = 'S' THEN 20
            WHEN r.codfruta IS NOT NULL AND r.NRROMANEIO IS NOT NULL  AND NVL(r.PROVISIONAR, 'N') = 'S' THEN 21
            WHEN r.codfruta IS NOT NULL AND NVL(r.PROVISIONAR, 'N') = 'S'  THEN 22
            WHEN r.codfruta IS NOT NULL AND r.MERCADO IS NOT NULL  AND NVL(r.PROVISIONAR, 'N') = 'S' THEN 23
            ELSE 24 -- Regra mais genérica (quando r.codproj = P_CODPROJ)
        END ASC
    FETCH FIRST 1 ROW ONLY; -- Traz apenas a primeira linha

    RETURN COALESCE(V_VLRPROVFORN, 0);
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RETURN 0;
END;
/
