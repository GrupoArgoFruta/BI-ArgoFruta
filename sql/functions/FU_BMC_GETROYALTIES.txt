CREATE OR REPLACE FUNCTION         FU_BMC_GETROYALTIES(
    P_DTREF                 DATE,
    P_CODPARCFORNECEDOR     INT,
    P_CODGRUPOPRODVARIEDADE INT,
    P_CODPARCPATENTE        INT,
    P_VLRRECEITALIQ         NUMBER,
    P_MERCADO              VARCHAR
) RETURN FLOAT DETERMINISTIC RESULT_CACHE RELIES_ON (AD_CONFROYALT, AD_CONFROYALTREG) IS
    V_VLRROYALTIES FLOAT;
BEGIN
    SELECT ROUND(
               (((P_VLRRECEITALIQ * NVL(r.PERCRECEITALIQUIDA, 100)) / 100) * r.PERCROYALTIES) / 100,
               2
           )
    INTO V_VLRROYALTIES
    FROM AD_CONFROYALT c
    LEFT JOIN AD_CONFROYALTREG r
        ON c.CODCONFROYALT = r.CODCONFROYALT
        AND TRUNC(P_DTREF) BETWEEN TRUNC(c.DTVIGENCIAINICIO) AND TRUNC(c.DTVIGENCIAFINAL)
        AND (r.CODPARCPATENTE = P_CODPARCPATENTE OR r.CODPARCPATENTE IS NULL)
        AND (r.CODGRUPOPROD = P_CODGRUPOPRODVARIEDADE OR r.CODGRUPOPROD IS NULL)
        AND (r.CODPARCFORNECEDOR = P_CODPARCFORNECEDOR OR r.CODPARCFORNECEDOR IS NULL)
        AND (r.MERCADO = P_MERCADO OR r.MERCADO IS NULL)
    ORDER BY
        CASE
            WHEN r.CODPARCPATENTE IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.CODPARCFORNECEDOR IS NOT NULL AND r.MERCADO IS NOT NULL THEN 1
            WHEN r.CODPARCPATENTE IS NOT NULL AND r.CODGRUPOPROD IS NOT NULL AND r.CODPARCFORNECEDOR IS NOT NULL THEN 2
            WHEN r.CODPARCPATENTE IS NOT NULL AND r.MERCADO IS NOT NULL THEN 3
            WHEN r.CODPARCPATENTE IS NOT NULL THEN 4
            ELSE 5
        END,
        PERCROYALTIES ASC
    FETCH FIRST 1 ROW ONLY;

    RETURN COALESCE(V_VLRROYALTIES, 0);
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RETURN 0;
END;
