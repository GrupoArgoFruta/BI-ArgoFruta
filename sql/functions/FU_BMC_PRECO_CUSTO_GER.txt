CREATE OR REPLACE FUNCTION         FU_BMC_PRECO_CUSTO_GER (
  P_DTREF        DATE,
  P_CULTIVAR     VARCHAR,
  P_MERCADO      VARCHAR,
  P_CODGRUPOPROD INT,
  P_TRANSPORTE   VARCHAR,
  P_TIPOCAIXA    FLOAT,
  P_CODPARCMP    INT,
  P_NROROMANEIO  VARCHAR,
  P_LOC_ENT      INT
)
 RETURN FLOAT RESULT_CACHE RELIES_ON (AD_CUSTOGER, AD_CUSTOGERITE) IS
  V_PRECO FLOAT := 0;
BEGIN

  BEGIN
    SELECT cgi.PRECO
      INTO V_PRECO
      FROM AD_CUSTOGER cg
      INNER JOIN AD_CUSTOGERITE cgi ON cg.CODCUSTOGER = cgi.CODCUSTOGER
     WHERE TRUNC(P_DTREF) BETWEEN cg.DATAINICIAL AND cg.DATAFINAL
       AND cgi.CULTIVAR = P_CULTIVAR
       AND cgi.TIPOCAIXA = P_TIPOCAIXA
       AND (cgi.CODGRUPOPROD IS NULL OR cgi.CODGRUPOPROD = P_CODGRUPOPROD)
       AND (cgi.TRANSPORTE IS NULL OR cgi.TRANSPORTE = P_TRANSPORTE)
       AND (cgi.MERCADO IS NULL OR cgi.MERCADO = P_MERCADO)
       AND (cgi.CODPARCMP IS NULL OR cgi.CODPARCMP = P_CODPARCMP)
       AND (cgi.NRROMANEIO IS NULL OR cgi.NRROMANEIO = P_NROROMANEIO)
       AND (cgi.CODLOCAL IS NULL OR cgi.CODLOCAL = P_LOC_ENT)
       AND cgi.ATIVO = 'S'
     ORDER BY
       CASE WHEN cgi.CODLOCAL IS NOT NULL THEN 1 ELSE 0 END +
       CASE WHEN cgi.CODGRUPOPROD IS NOT NULL THEN 1 ELSE 0 END +
       CASE WHEN cgi.TRANSPORTE IS NOT NULL THEN 1 ELSE 0 END +
       CASE WHEN cgi.MERCADO IS NOT NULL THEN 1 ELSE 0 END +
       CASE WHEN cgi.CODPARCMP IS NOT NULL THEN 1 ELSE 0 END +
       CASE WHEN cgi.NRROMANEIO IS NOT NULL THEN 1 ELSE 0 END DESC,
       cgi.CODCUSTOGERITE DESC
     FETCH FIRST 1 ROW ONLY;
  EXCEPTION
    WHEN NO_DATA_FOUND THEN
      V_PRECO := 0;
  END;

  RETURN (V_PRECO);
END;
/
