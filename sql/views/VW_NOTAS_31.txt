ON (
TOP.CODTIPOPER = CAB.CODTIPOPER AND
      CAB.DHTIPOPER = TOP.DHALTER
)
      
AND COALESCE(TOP.AD_CRE_DEB,
'X') ='X'

INNER 
JOIN
TGFNAT NAT
      
ON NAT.CODNAT = CAB.CODNAT
INNER 
JOIN
TSICUS CEN
      
ON CEN.CODCENCUS = CAB.CODCENCUS
LEFT

AD_FRUTA FRU
      
ON FRU.CODFRUTA = CEN.AD_CODFRUTA

TCSPRJ PRJ

TCSPRJ PRJ2

TSIMOE MOE

TGFCOM COM

TGFVEI NAV

NAV.CODVEICULO = CAB.AD_EX_NAVIO
)

AD_PORTO PO1

)
LEFT 
JOIN
AD_PORTO PO2
      
ON (
PO2.IDPORTO = CAB.AD_EX_PORTODESCARGA
)   
LEFT 
JOIN
VW_BMC_GET_QTD_DEV_VENDA_ROM DEV 
      
ON  DEV.nunotaorig=CAB.NUNOTA 
      
AND DEV.sequenciaorig=ITE.SEQITE
      
AND DEV.ROMANEIO=ITE.ROMANEIO

JOIN
DEV_ITEM DEVO

AND DEVO.sequenciaorig=ITEO.SEQUENCIA 
LEFT 
JOIN
QTDNEG_POR_NUNOTA QNOTA

LEFT 
JOIN
QTDNEG_POR_CENCUS QCEN 
ON QCEN.CODCENCUS = CAB.CODCENCUS

JOIN
QTDNEG_POR_CALIBRE QCAL 
ON QCAL.NUNOTA=ITE.NUNOTA 
AND QCAL.AD_CALIBRE=ITE.CODCALIBRE 
AND QCAL.CODPROD=ITE.PRODUCTCODE

JOIN
QTDNEG_POR_CONTROLE QCTR 
ON QCTR.NUNOTA=ITE.NUNOTA 
AND QCTR.CONTROLE=ITE.CONTROLE 
AND QCTR.CODPROD=ITE.PRODUCTCODE

JOIN
QTDNEG_POR_FORN QFOR

JOIN
VLRPROVFORN_POR_CENCUS QPRV

LEFT

PARC_PATENTE QPAT

AND CAB.DTNEG >= QPAT.DTVIGENCIAINICIO 
AND CAB.DTNEG < QPAT.DTVIGENCIAFIM + 1
LEFT 
JOIN
VARIEDADE_MP QVAR

LEFT 
JOIN
NF_ENT_MP QNFMP 
ON QNFMP.NROUNICO =  ITE.ROMANEIO_NUM
LEFT 
JOIN
AD_ROMANEIOENTR ROM1 
ON ROM1.NROUNICO= ITE.ROMANEIO_NUM
LEFT 
JOIN
VW_PERCPROC_NF_V4 PER_NF 
ON CAB.NUNOTA=PER_NF.NUNOTA 
AND COALESCE(prj2.codproj,
prj.codproj)=PER_NF.PROCESSO 
LEFT 
JOIN
MV_AD_REC_COMD_NUNOTA OPN  
ON OPN.NUNOTA = CAB.NUNOTA -- 
LEFT JOIN
VW_AD_REC_COMD OPN 
ON OPN.NUNOTA=CAB.NUNOTA  
LEFT 
JOIN
(
SELECT
SUM(CVLRMOEDAEX) CVLRMOEDAEX,
SUM(DVLRMOEDAEX) DVLRMOEDAEX,
SUM(CVLRNOTA) CVLRNOTA,
SUM(DVLRNOTA) DVLRNOTA,
CODPROJ CODPROJETO 
		   FROM
			(SELECT
CASE 
WHEN DC.AD_CRE_DEB ='C' THEN (DC.VLRMOEDAEX) 
ELSE 0 
END CVLRMOEDAEX,

				  CASE 
WHEN DC.AD_CRE_DEB ='D' THEN (DC.VLRMOEDAEX) 
ELSE 0 
END DVLRMOEDAEX,

				  CASE 
WHEN DC.AD_CRE_DEB ='C' THEN (DC.VLRNOTA) 
ELSE 0 
END CVLRNOTA,

				  CASE 
WHEN DC.AD_CRE_DEB ='D' THEN (DC.VLRNOTA) 
ELSE 0 
END DVLRNOTA,

				  DC.CODPROJ
			FROM VW_ARG_CRE_DEB DC
			WHERE CODPROJ<>0 
AND AD_CONSCTRL='N')  CREDEB1 
		    --
WHERE
CODPROJ='4150240110'
			GROUP 
BY
CODPROJ) CRDB 
ON CRDB.CODPROJETO= COALESCE(PRJ2.CODPROJ,
PRJ.CODPROJ)
LEFT 
JOIN
TGFPRO PROMPCX
  
ON PROMPCX.CODPROD = ITE.CODPRODCX	
LEFT 
JOIN
VW_DESCFIN DFIN 
  
ON DFIN.NUNOTA=ITEO.NUNOTA 
  
AND DFIN.SEQITE=ITEO.SEQUENCIA 
  
AND DFIN.QUANTITY=COALESCE(ITE.QUANTITY,
0)
  
AND DFIN.TRACEABILITY=ITE.TRACEABILITY  
LEFT 
JOIN
PED_FRETE_VENDAS  PED_FR 
ON PED_FR.CODPROJ=ITE.CODPROJ  
LEFT 
JOIN
PED_DESP_LOG PED_LOG 
ON PED_LOG.CODPROJ=ITE.CODPROJ  
LEFT 
JOIN
TSICID CIDCO
      
ON CIDCO.CODCID = CASE 
WHEN FRU.NOME LIKE 'UVA%' 
OR FRU.NOME LIKE 'MANG%' THEN 3767 
ELSE NVL(CAB.AD_CODCIDCOL,
3767) END
LEFT 
JOIN
TSICID CIDDE
      
ON CIDDE.CODCID = CASE 
    WHEN PO1.IDPORTO IS NULL THEN PARCLI.CODCID 
    ELSE PO1.CODCID 
END   
LEFT 
JOIN
(
SELECT
    MAX(FHST.VLRCTE) VLRCTE,
    FHST.CODCIDINICTE,
    FHST.CODCIDFIMCTE,
    FHST.CIF_FOB 
FROM
    TB_FRETES_HST FHST 
GROUP BY
    FHST.CODCIDINICTE,
    FHST.CODCIDFIMCTE,
    FHST.CIF_FOB
) FHST 
  
ON FHST.CODCIDINICTE = CASE 
       
    WHEN CAB.CODTIPOPER = 1161 THEN NULL 
       
    ELSE CASE 
        WHEN FRU.NOME LIKE 'UVA%' 
        OR FRU.NOME LIKE 'MANG%' 
            THEN 3767 
        ELSE NVL(CAB.AD_CODCIDCOL,
    3767) 
END 
     
END --
AND CAB.CODPROJ<>0
  
AND FHST.CODCIDFIMCTE = CASE 
WHEN PO1.IDPORTO IS NULL THEN PARCLI.CODCID 
ELSE PO1.CODCID END
  
AND FHST.CIF_FOB = CAB.CIF_FOB
LEFT 
JOIN
(
    SELECT
        MAX(FMAHST.VLRTOT) VLRTOT,
        FMAHST.AD_EX_PORTOEMBARQUE,
        FMAHST.AD_EX_PORTODESCARGA 
    FROM
        TB_FRETES_MARITIMO_HST FMAHST 
    GROUP BY
        FMAHST.AD_EX_PORTOEMBARQUE,
        FMAHST.AD_EX_PORTODESCARGA
) FMAHST 
    ON FMAHST.AD_EX_PORTOEMBARQUE=CAB.AD_EX_PORTOEMBARQUE 
    AND FMAHST.AD_EX_PORTODESCARGA = CAB.AD_EX_PORTODESCARGA  
    AND CAB.AD_INCOTERM IN (
        'CIF',
    'CIP',
    'CPT')                              
LEFT 
JOIN
    VW_M_CUSTOMED_SEMANA MSEM 
        ON TO_CHAR((ROM1.DTENTRADA ),
    'IW-YYYY')=MSEM.SEMANA_ENTR 
    AND ITE.CODPRODMP=MSEM.CODPROD 
    AND (ITE.CALIBER=MSEM.CALIBRE 
    OR 'MG'=MSEM.CALIBRE)
LEFT 
JOIN
    (
        
		    SELECT
		        CODPROD,
		        CONTROLEPA,
		        -- Quantidade real: zera tudo que for COMPLEMENTAR (independente de vir 0 ou 1)
		        SUM(CASE 
            WHEN UPPER(DESCROPER) LIKE '%COMPLEMENTAR%' THEN 0
		                 
            ELSE QTDNEG
		            
        END)                                                  AS QTDNEG_REAL,
		        -- Valor total: soma original + todos os complementos
		        SUM(QTDNEG * VLRUNIT)                                     AS VLRTOTAL,
		        -- Unitário real recalculado pela QTD verdadeira
		        ROUND(
		            SUM(QTDNEG * VLRUNIT)
		            / NULLIF(
		                SUM(CASE 
            WHEN UPPER(DESCROPER) LIKE '%COMPLEMENTAR%' THEN 0
		                         
            ELSE QTDNEG
		                    
        END), 0)
		        , 4)                                                      AS VLRUNIT,
		        -- Flags de auditoria
		        SUM(CASE 
            WHEN UPPER(DESCROPER) LIKE '%COMPLEMENTAR%'
		                 THEN QTDNEG * VLRUNIT 
            ELSE 0 
        END)                AS VLR_COMPLEMENTO,
		        MAX(CASE 
            WHEN UPPER(DESCROPER) LIKE '%COMPLEMENTAR%'
		                 THEN 'S' 
            ELSE 'N' 
        END)                           AS TEM_COMPLEMENTO
		    
    FROM
        ARG_COMPRA_MP
		    
    GROUP BY
        CODPROD,
        CONTROLEPA
		) MPE 
            ON ITE.PRODUCTCODE=MPE.CODPROD 
            AND ITE.CONTROLE=MPE.CONTROLEPA
LEFT 
    JOIN
        ARG_FRETES_RURALVD FRRU 
            ON FRRU.CODPROJ = ITE.CODPROJ  
LEFT 
    JOIN
        VW_ARG_DEB_CRE_ITE CD_ITE 
            ON CD_ITE.CODPROJ=COALESCE(PRJ2.CODPROJ,
        PRJ.CODPROJ) 
        AND CD_ITE.CODPROD=ITE.PRODUCTCODE 
        AND CD_ITE.CONTROLE=ITE.CONTROLE 
        AND CD_ITE.CT_CL='CT' 
        AND CD_ITE.AD_CONSCTRL='S'
LEFT 
    JOIN
        VW_ARG_DEB_CRE_ITE CD_ITE2 
            ON CD_ITE2.CODPROJ=COALESCE(PRJ2.CODPROJ,
        PRJ.CODPROJ) 
        AND CD_ITE2.CODPROD=ITE.PRODUCTCODE 
        AND CD_ITE2.AD_CALIBRE=ITE.CODCALIBRE 
        AND CD_ITE2.CT_CL='CL' 
        AND CD_ITE2.AD_CONSCTRL='C'
LEFT 
    JOIN
        TIPO_PARCERIA_LKP TPL
       
            ON TPL.CODTIPOPARCERIA = PARPROD.AD_TIPOPARCERIA
WHERE NVL(ITEO.AD_QTDNEGOR,
        ITEO.QTDNEG) - NVL(DEVO.V_QTDDEV,
        0) > 0   --    
        AND CAB.NUNOTA=157921
  
        AND NOT EXISTS (SELECT
            codproj 
        FROM
            AD_NOTASEXC ex 
        WHERE
            ex.codproj=ITE.CODPROJ)
  
        AND NOT EXISTS (SELECT
            nunota 
        FROM
            AD_NOTASEXC ex 
        WHERE
            ex.nunota=cab.nunota)
  
        AND NOT EXISTS (SELECT
            CODPARC 
        FROM
            VW_TGFPARC_TGFEMP EP  
        WHERE
            EP.CODPARC= CAB.CODPARC)
  
        AND (CAB.AD_NUNOTACREF IS NULL)
  
        AND NOT EXISTS (SELECT
            nunota 
        FROM
            tgfcab ex1 
        WHERE
            ex1.AD_NUNOTASUB=cab.nunota)
