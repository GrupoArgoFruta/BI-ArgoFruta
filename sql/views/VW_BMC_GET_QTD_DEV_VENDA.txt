    SELECT
        SUM(i.qtdneg) V_QTDDEV
         ,
        SUM(i.vlrtot) V_VLRDEV
         ,
        v.nunotaorig
         ,
        v.sequenciaorig
         ,
        CASE
            WHEN c.codtipoper NOT IN (1227,
            1228) THEN 'N'
            ELSE 'S'
        END REFUGO
         ,
        listagg(v.NUNOTA,
        ',') nunotadest,

         'D' AS TIPO,

         listagg(c.chavenfe,
        ',') chavenfe

    FROM
        TGFVAR v

    INNER JOIN
        TGFCAB c

            ON c.nunota = v.nunota

            AND c.tipmov = 'D'

            AND C.STATUSNOTA = 'L'

    INNER JOIN
        TGFITE i

            ON i.nunota = c.nunota

            AND i.sequencia = v.sequencia
GROUP
    BY
        v.nunotaorig
        ,
        v.sequenciaorig
        ,
        CASE
            WHEN c.codtipoper NOT IN (1227,
            1228) THEN 'N'
            ELSE 'S'
        END,
        v.NUNOTA,
        c.chavenfe
