    SELECT

    CAB.CODEMP,

    CASE
            WHEN PRJI.CODPROJ=0 THEN PRJC.CODPROJ
            ELSE PRJI.CODPROJ
        END CODPROJ,

    CASE
            WHEN PRJI.CODPROJ=0 THEN  PRJC.IDENTIFICACAO
            ELSE PRJI.IDENTIFICACAO
        END PROJETO,

    CAB.NUNOTA,

    CAB.NUMNOTA,

    CAB.DTENTSAI,

    CAB.DTNEG,

    CAB.DTFATUR,

    CAB.DTVAL,

    CAB.DTCONTAB,

    PAR.CODPARC,

    PAR.NOMEPARC,

    CAB.CIF_FOB,

    CAB.CODMOEDA,

    CASE
            WHEN CUSI.CODCENCUS=0 THEN CUSC.CODCENCUS
            ELSE CUSI.CODCENCUS
        END CODCENCUS,

    CASE
            WHEN CUSI.CODCENCUS=0 THEN CUSC.DESCRCENCUS
            ELSE CUSI.DESCRCENCUS
        END DESCRCENCUS,

    CAB.CODNAT,

    NAT.DESCRNAT,

    CAB.VLRNOTA,

    CAB.VLRFRETE,

    CAB.VLRMOEDA ,

    CAB.TIPMOV,

    CAB.CODTIPOPER,

    CAB.CHAVENFE,

    CAB.STATUSNOTA,

    CAB.AD_NROROMANEIO,

    CAB.OBSERVACAO ,

    TOP.DESCROPER,

    ITE.SEQUENCIA ,

    ITE.CODPROD,

    PRO.DESCRPROD,

    ITE.USOPROD USOPROD_ITE,

    PRO.USOPROD USOPROD_PRO,

    ITE.ATUALESTOQUE ,

    ITE.CONTROLE,

    ITE.AD_CONTROLE,

    ITE.AD_CONTROLEALT,

    ITE.AD_PALLET,

    ITE.AD_NROPALLET,

    ITE.AD_NROPALLETEX,

    ITE.AD_CALIBRE,

    ITE.CODLOCALORIG,

    ITE.CODLOCALTERC,

    ITE.AD_LOCALDEST,

    ITE.AD_CODLOCAL,

    ITE.AD_CODLOCAL2 ,

    ITE.CODVOL,

    ITE.QTDNEG,

    ITE.VLRUNIT,

    ITE.VLRTOT,

    ITE.VLRUNITMOE,

    ITE.VLRTOTMOE,

    ITE.AD_ROMANEIOENT,

    ITE.CODCFO,

    --ITE.AD_CODPROJ,

    PRO.PESOBRUTO,

    PRO.PESOLIQ ,

    GRU.CODGRUPOPROD ,

    GRU.DESCRGRUPOPROD ,

    GRU.VALEST,

    ITE.AD_CODPARVALVULA,

    ITE.AD_CODAREA,

    ITE.VLRDESC,

    PRO.AD_CULTIVAR,

    CAB.PENDENTE,

    CAB.CODUSUINC,

    USU.NOMEUSU NOMEUSUINC,

    CASE
         WHEN UFS.CODPAIS = 55
        AND  CAB.AD_EX_DTPREVEMB IS NULL
         THEN 'MI'

        ELSE 'ME'

    END AS MERCADO,

    CAB.CODCIDINICTE,

    CAB.CODCIDFIMCTE,

    CAB.CHAVECTE,

    CAB.STATUSNFE,

    CAB.CODPARCDEST,

    CAB.AD_INCOTERM,

    UPPER(TOP.GRUPO) TOP_GRUPO,

    CAB.AD_STCUTOFF,

    ROUND(CAB.VLRNOTA/CASE
        WHEN CAB.VLRMOEDA=0 THEN 1
        ELSE CAB.VLRMOEDA
    END,
    2) VLRNOTA_MOE,

    CASE
        WHEN CAB.VLRMOEDA=0 THEN 1
        ELSE CAB.VLRMOEDA
    END VLRCOT,

    TOP.AD_CRE_DEB,

    CAB.AD_CONSCTRL,

    ITE.AD_QTDNEG,

    ITE.AD_QTDNEGOR
FROM
    TGFCAB CAB
INNER
JOIN
    TGFTOP TOP ON
    CAB.CODTIPOPER = TOP.CODTIPOPER

    AND CAB.DHTIPOPER = TOP.DHALTER
INNER
JOIN
    TGFITE ITE ON
    CAB.NUNOTA = ITE.NUNOTA
INNER
JOIN
    TGFPRO PRO ON
     ITE.CODPROD = PRO.CODPROD
INNER
JOIN
    TGFGRU GRU ON
    PRO.CODGRUPOPROD = GRU.CODGRUPOPROD
INNER
JOIN
    TGFPAR PAR ON
    CAB.CODPARC = PAR.CODPARC
INNER
JOIN
    TCSPRJ PRJI ON
    PRJI.CODPROJ = NVL(ITE.AD_CODPROJ,0)
INNER
JOIN
    TCSPRJ PRJC ON
    PRJC.CODPROJ = NVL(CAB.CODPROJ,0)
INNER
JOIN
    TGFNAT NAT
      ON
    NAT.CODNAT=NVL(CAB.CODNAT,
    0)
INNER
JOIN
    TSICUS CUS
        ON
    CUS.CODCENCUS=NVL(CAB.CODCENCUS,
    0)
INNER
JOIN
    TSICID CID

        ON CID.CODCID = PAR.CODCID
INNER
JOIN
    TSIUFS UFS

        ON UFS.CODUF = CID.UF
INNER
JOIN
    TSIPAI PAI

        ON PAI.CODPAIS = UFS.CODPAIS
INNER
JOIN
    TSICUS CUSI ON
    CUSI.CODCENCUS = NVL(ITE.AD_CODCENCUS,0)
INNER
JOIN
    TSICUS CUSC ON
    CUSC.CODCENCUS = NVL(CAB.CODCENCUS,0)
LEFT
JOIN
    TSIUSU USU
        ON
    USU.CODUSU = CAB.CODUSUINC
